# v1.0.6
> 支持 observe 一个 path.

```js
@observe
class MyBusinessComponent extends React.Component {
  $memory = {
    'path.subpath': {
      get: function() {
        return this._subpath;
      },
      set: function(inValue) {
        this._subpath = inValue;
      }
    }
  };
  render() {}
}
```
